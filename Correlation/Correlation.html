\[r = \frac{1}{n-1}\sum_{i=1}^{n}\frac{x_i - \bar{x}}{s_x}\cdot \frac{y_i - \bar{y}}{s_y}\]

x <- signdist$Age
y <- signdist$Distance
xbar <- mean(x)
sx <- sd(x)
ybar <- mean(y)
sy <- sd(y)
xMINUSxbar <- x - xbar
yMINUSybar <- y - ybar
zx <- xMINUSxbar/sx
zy <- yMINUSybar/sy
r <- 1/(dim(signdist)[1] - 1)*sum(zx*zy)
c(xbar, sx, ybar, sy, r)
